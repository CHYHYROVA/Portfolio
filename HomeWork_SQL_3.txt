SELECT department_id, COUNT(employee_id) AS count_id
FROM employees 
WHERE department_id IN (SELECT department_id
                       FROM job_history
                       GROUP BY department_id
                       HAVING COUNT(department_id)>1) 
GROUP BY department_id                       
ORDER BY count_id DESC;
